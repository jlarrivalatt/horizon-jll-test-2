
!declare the doctype to tell browser what version of HTML
<!DOCTYPE html>
!<htlm> is the root element
! lang specifies the language of text content
<html lang="en">
    !head element fro important meta-information about our webpages
    <head>
        !always include meta charset element so special symbols and characters from diff languages appear correctly
        <meta charset="UTF-8">
        !always include title element which is displayed in webpage's browser tab
        <title>My First Webpage</title>
    </head>
    !body for content to be displayed, text, images, links
    <body>
        !heading
        <h1>Hello World!</h1>
    </body>
</html>
